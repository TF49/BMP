from typing import Optional, Any


class AgentException(Exception):
    """Agent 基础异常"""

    def __init__(
        self,
        message: str,
        code: str = "AGENT_ERROR",
        details: Optional[dict] = None,
        trace_id: Optional[str] = None
    ):
        self.message = message
        self.code = code
        self.details = details or {}
        self.trace_id = trace_id
        super().__init__(self.message)


class ConfigurationError(AgentException):
    """配置错误"""

    def __init__(self, message: str, details: Optional[dict] = None):
        super().__init__(message, code="CONFIG_ERROR", details=details)


class ModelError(AgentException):
    """模型错误"""

    def __init__(self, message: str, details: Optional[dict] = None):
        super().__init__(message, code="MODEL_ERROR", details=details)


class ToolError(AgentException):
    """工具错误"""

    def __init__(self, message: str, details: Optional[dict] = None):
        super().__init__(message, code="TOOL_ERROR", details=details)


class SessionError(AgentException):
    """会话错误"""

### 4.3 LLM 基类 (app/llm/base.py)

```python
from abc import ABC, abstractmethod
from typing import Optional, Dict, Any
from langchain_core.language_models import BaseChatModel
from langchain_core.messages import BaseMessage


class BaseLLMAdapter(ABC):
    """LLM 适配器基类"""

    def __init__(self, model_name: str, **kwargs):
        self.model_name = model_name
        self.kwargs = kwargs

    @abstractmethod
    def get_model(self) -> BaseChatModel:
        """获取 LangChain 模型实例"""
        pass

    @abstractmethod
    async def generate(
        self,
        messages: list[BaseMessage],
        **kwargs
    ) -> str:
        """生成文本"""
        pass

    @abstractmethod
    async def generate_structured(
        self,
        messages: list[BaseMessage],
        schema: Dict[str, Any],
        **kwargs
    ) -> Dict[str, Any]:
        """生成结构化输出"""
        pass
```

### 4.4 OpenAI 适配器 (app/llm/openai_adapter.py)

```python
from langchain_openai import ChatOpenAI
from langchain_core.messages import BaseMessage
from typing import Dict, Any
from .base import BaseLLMAdapter


class OpenAIAdapter(BaseLLMAdapter):
    """OpenAI 模型适配器"""

    def __init__(
        self,
        api_key: str,
        base_url: str,
        model_name: str,
        **kwargs
    ):
        super().__init__(model_name, **kwargs)
        self.api_key = api_key
        self.base_url = base_url
        self._model = None

    def get_model(self):
        if self._model is None:
            self._model = ChatOpenAI(
                api_key=self.api_key,
                base_url=self.base_url,
                model=self.model_name,
                **self.kwargs
            )
        return self._model

    async def generate(
        self,
        messages: list[BaseMessage],
        **kwargs
    ) -> str:
        model = self.get_model()
        response = await model.ainvoke(messages, **kwargs)
        return response.content

    async def generate_structured(
        self,
        messages: list[BaseMessage],
        schema: Dict[str, Any],
        **kwargs
    ) -> Dict[str, Any]:
        model = self.get_model()
        # 使用 with_structured_output
        structured_model = model.with_structured_output(schema)
        response = await structured_model.ainvoke(messages, **kwargs)
        return response
```

### 4.5 结构化日志 (app/observability/logging.py)

```python
import logging
import json
import sys
from datetime import datetime
from typing import Any, Dict
from app.core.config import settings


class JSONFormatter(logging.Formatter):
    """JSON 格式化器"""

    def format(self, record: logging.LogRecord) -> str:
        log_data = {
            "timestamp": datetime.utcnow().isoformat(),
            "level": record.levelname,
            "logger": record.name,
            "message": record.getMessage(),
            "module": record.module,
            "function": record.funcName,
            "line": record.lineno,
        }

        if hasattr(record, "trace_id"):
            log_data["trace_id"] = record.trace_id

        if hasattr(record, "user_id"):
            log_data["user_id"] = record.user_id

        if hasattr(record, "extra"):
            log_data.update(record.extra)

        if record.exc_info:
            log_data["exception"] = self.formatException(record.exc_info)

        return json.dumps(log_data, ensure_ascii=False)


def setup_logging():
    """配置日志"""
    log_level = getattr(logging, settings.log_level.upper(), logging.INFO)

    # 创建根日志记录器
    root_logger = logging.getLogger()
    root_logger.setLevel(log_level)

    # 控制台处理器
    console_handler = logging.StreamHandler(sys.stdout)
    console_handler.setLevel(log_level)

    if settings.log_format == "json":
        console_handler.setFormatter(JSONFormatter())
    else:
        console_handler.setFormatter(
            logging.Formatter(
                "%(asctime)s - %(name)s - %(levelname)s - %(message)s"
            )
        )

    root_logger.addHandler(console_handler)

    return root_logger


def get_logger(name: str) -> logging.Logger:
    """获取日志记录器"""
    return logging.getLogger(name)
```

import uuid
from contextvars import ContextVar
from typing import Optional

# ContextVar 用于存储当前请求的 trace_id
_trace_id: ContextVar[Optional[str]] = ContextVar("trace_id", default=None)


def generate_trace_id() -> str:
    """生成 trace_id"""
    return str(uuid.uuid4())


def set_trace_id(trace_id: str) -> None:
    """设置当前请求的 trace_id"""
    _trace_id.set(trace_id)


def get_trace_id() -> Optional[str]:
    """获取当前请求的 trace_id"""
    return _trace_id.get()


def get_or_create_trace_id() -> str:
    """获取或创建 trace_id"""
    trace_id = get_trace_id()
    if trace_id is None:
        trace_id = generate_trace_id()
        set_trace_id(trace_id)
    return trace_id

### 4.7 会话管理 (app/memory/session.py)

```python
from typing import Optional, Dict, Any
from datetime import datetime, timedelta
from app.core.exceptions import SessionError
from app.observability.tracing import get_trace_id
import uuid


class SessionManager:
    """会话管理器"""

    def __init__(self):
        # 开发环境使用内存存储，生产环境使用 PostgreSQL
        self._sessions: Dict[str, Dict[str, Any]] = {}

    async def create_session(
        self,
        user_id: str,
        agent_type: str,
        venue_id: Optional[str] = None
    ) -> Dict[str, Any]:
        """创建会话"""
        conversation_id = f"conv_{uuid.uuid4().hex[:12]}"
        thread_id = str(uuid.uuid4())
        expires_at = datetime.utcnow() + timedelta(hours=1)

        session = {
            "conversation_id": conversation_id,
            "user_id": user_id,
            "agent_type": agent_type,
            "thread_id": thread_id,
            "venue_id": venue_id,
            "status": "active",
            "created_at": datetime.utcnow().isoformat(),
            "expires_at": expires_at.isoformat(),
            "trace_id": get_trace_id(),
        }

        self._sessions[conversation_id] = session
        return session

    async def get_session(self, conversation_id: str) -> Optional[Dict[str, Any]]:
        """获取会话"""
        session = self._sessions.get(conversation_id)
        if session is None:
            return None

        # 检查是否过期
        expires_at = datetime.fromisoformat(session["expires_at"])
        if datetime.utcnow() > expires_at:
            del self._sessions[conversation_id]
            return None

        return session

    async def delete_session(self, conversation_id: str) -> bool:
        """删除会话"""
        if conversation_id in self._sessions:
            del self._sessions[conversation_id]
            return True
        return False

    async def cleanup_expired_sessions(self) -> int:
        """清理过期会话"""
        expired_count = 0
        now = datetime.utcnow()

        for conversation_id, session in list(self._sessions.items()):
            expires_at = datetime.fromisoformat(session["expires_at"])
            if now > expires_at:
                del self._sessions[conversation_id]
                expired_count += 1

        return expired_count


# 全局会话管理器实例
session_manager = SessionManager()
```

### 4.8 Mock Agent (app/agents/mock_agent.py)

```python
from typing import Dict, Any, Optional
from langgraph.graph import StateGraph, END
from app.observability.logging import get_logger
from app.observability.tracing import get_trace_id


logger = get_logger(__name__)


class MockAgentState:
    """Mock Agent 状态"""

    messages: list[Dict[str, Any]]
    user_input: Optional[str] = None
    response: Optional[str] = None


class MockAgent:
    """测试用 Mock Agent"""

    def __init__(self):
        self.graph = self._build_graph()

    def _build_graph(self) -> StateGraph:
        """构建状态图"""
        workflow = StateGraph(MockAgentState)

        # 添加节点
        workflow.add_node("process_input", self._process_input)
        workflow.add_node("generate_response", self._generate_response)

        # 设置边
        workflow.set_entry_point("process_input")
        workflow.add_edge("process_input", "generate_response")
        workflow.add_edge("generate_response", END)

        return workflow.compile()

    async def _process_input(self, state: MockAgentState) -> MockAgentState:
        """处理用户输入"""
        logger.info(f"Processing input: {state.user_input}", trace_id=get_trace_id())
        state["messages"] = state.get("messages", [])
        state["messages"].append({
            "role": "user",
            "content": state["user_input"]
        })
        return state

    async def _generate_response(self, state: MockAgentState) -> MockAgentState:
        """生成响应"""
        user_input = state["user_input"]
        response = f"我收到了您的消息：{user_input}。这是一个 Mock Agent 响应。"
        
        logger.info(f"Generating response: {response}", trace_id=get_trace_id())
        
        state["response"] = response
        state["messages"].append({
            "role": "assistant",
            "content": response
        })
        
        return state

    async def run(self, user_input: str) -> Dict[str, Any]:
        """运行 Agent"""
        initial_state = {
            "user_input": user_input,
            "messages": [],
            "response": None
        }

        trace_id = get_trace_id()
        logger.info(f"Starting Mock Agent with trace_id: {trace_id}")

        try:
            final_state = await self.graph.ainvoke(initial_state)
            return {
                "response": final_state["response"],
                "messages": final_state["messages"],
                "trace_id": trace_id
            }
        except Exception as e:
            logger.error(f"Mock Agent error: {e}", trace_id=trace_id)
            raise
```

## 9. 常见问题

### 9.1 依赖安装失败

**问题**：pip install 失败

**解决方案**：
```bash
# 使用国内镜像
pip install -r requirements.txt -i https://pypi.tuna.tsinghua.edu.cn/simple

# 升级 pip
python -m pip install --upgrade pip
```

### 9.2 模型连接失败

**问题**：OpenAI API 连接失败

**解决方案**：
- 检查 API Key 是否正确
- 检查网络连接
- 检查 base_url 配置
- 使用代理或国内替代服务

### 9.3 数据库连接失败

**问题**：PostgreSQL 连接失败

**解决方案**：
- 开发环境可使用 SQLite
- 检查数据库服务是否启动
- 检查连接字符串配置
- 检查防火墙设置
